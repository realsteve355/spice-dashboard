// The fundamental Spice Economic Constant
export const ERGON_THRESHOLD_METRIC = (autoOutput, popSize, survivalCost) => {
  const totalCapacity = autoOutput.joules * autoOutput.efficiency;
  const communityNeeds = popSize * survivalCost;
  return totalCapacity / communityNeeds; // If > 1, the Spice Model is viable.
};
